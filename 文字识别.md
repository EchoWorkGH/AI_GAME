# 初始化，use_angle_cls=True 可以识别旋转 180 度的文字
ocr = PaddleOCR(use_angle_cls=True, lang="ch") 

img_path = 'test_chinese.jpg'
result = ocr.ocr(img_path, cls=True)

# 打印结果
for idx in range(len(result)):
    res = result[idx]
    for line in res:
        # line[0] 是四个角坐标，line[1][0] 是文本，line[1][1] 是置信度
        print(f"检测到文字: {line[1][0]} | 置信度: {line[1][1]:.2f}")
